- g++-4.4
- libx11-dev
- libxt-dev
- libxext-dev
- libfreetype6-dev
- libfontconfig1-dev
- chrpath
- libxrender-dev
- freeglut3-dev
- libgtk2.0-dev (to enable gtkstyle)
- libdbus-1-dev
