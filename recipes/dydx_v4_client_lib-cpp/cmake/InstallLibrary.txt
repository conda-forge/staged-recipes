set_target_properties(dydx_v4_client_lib PROPERTIES
  RUNTIME_OUTPUT_DIRECTORY "${CMAKE_INSTALL_PREFIX}/bin"
  LIBRARY_OUTPUT_DIRECTORY "${CMAKE_INSTALL_PREFIX}/lib"
  ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_INSTALL_PREFIX}/lib")

install(TARGETS dydx_v4_client_lib
    EXPORT dydx_v4_client_lib-config
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
    COMPONENT client)

if(WIN32)
  install(FILES $<TARGET_LINKER_FILE:dydx_v4_client_lib>
    DESTINATION "${CMAKE_INSTALL_PREFIX}/lib"
    COMPONENT client
  )
endif()
